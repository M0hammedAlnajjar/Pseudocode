Task 293: Shape Drawing App :



CLASS Shape
    ATTRIBUTES
        color : String
        x : Integer
        y : Integer

    CONSTRUCTOR Shape(color : String, x : Integer, y : Integer)
        SET color = color
        SET x = x
        SET y = y

    METHODS
        calculateArea() : Double
        calculatePerimeter() : Double
END CLASS


CLASS Circle INHERITS Shape
    ATTRIBUTES
        radius : Double

    CONSTRUCTOR Circle(color : String, x : Integer, y : Integer, radius : Double)
        CALL Shape(color, x, y)
        SET radius = radius

    METHODS
        calculateArea() : Double
            RETURN 3.14 * radius * radius

        calculatePerimeter() : Double
            RETURN 2 * 3.14 * radius
END CLASS


CLASS Rectangle INHERITS Shape
    ATTRIBUTES
        width : Double
        height : Double

    CONSTRUCTOR Rectangle(color : String, x : Integer, y : Integer,
                          width : Double, height : Double)
        CALL Shape(color, x, y)
        SET width = width
        SET height = height

    METHODS
        calculateArea() : Double
            RETURN width * height

        calculatePerimeter() : Double
            RETURN 2 * (width + height)
END CLASS


CLASS Triangle INHERITS Shape
    ATTRIBUTES
        side1 : Double
        side2 : Double
        side3 : Double

    CONSTRUCTOR Triangle(color : String, x : Integer, y : Integer,
                         side1 : Double, side2 : Double, side3 : Double)
        CALL Shape(color, x, y)
        SET side1 = side1
        SET side2 = side2
        SET side3 = side3

    METHODS
        calculateArea() : Double
            RETURN "Use Heron's Formula"

        calculatePerimeter() : Double
            RETURN side1 + side2 + side3
END CLASS


CLASS Canvas
    ATTRIBUTES
        shapes : List<Shape>

    CONSTRUCTOR Canvas()
        CREATE empty shapes list

    METHODS
        addShape(shape : Shape)
            ADD shape TO shapes

        displayShapes()
            FOR EACH shape IN shapes
                PRINT shape
            END FOR
END CLASS


START PROGRAM

DECLARE canvas : Canvas
SET canvas = NEW Canvas()

DECLARE circle : Circle
SET circle = NEW Circle("Red", 10, 20, 5)

DECLARE rectangle : Rectangle
SET rectangle = NEW Rectangle("Blue", 30, 40, 8, 4)

DECLARE triangle : Triangle
SET triangle = NEW Triangle("Green", 50, 60, 3, 4, 5)

CALL canvas.addShape(circle)
CALL canvas.addShape(rectangle)
CALL canvas.addShape(triangle)

PRINT "Circle Area = " + circle.calculateArea()
PRINT "Rectangle Area = " + rectangle.calculateArea()
PRINT "Triangle Perimeter = " + triangle.calculatePerimeter()

CALL canvas.displayShapes()

END PROGRAM



Task 294: Employee Payroll 




PROGRAM EmployeePayroll

CLASS Employee
    ATTRIBUTES
        name : String
        id : Integer
        hireDate : String

    CONSTRUCTOR Employee(name : String, id : Integer, hireDate : String)
        SET name = name
        SET id = id
        SET hireDate = hireDate

    METHODS
        calculateMonthlyPay() : Double
END CLASS


CLASS SalariedEmployee INHERITS Employee
    ATTRIBUTES
        monthlySalary : Double

    CONSTRUCTOR SalariedEmployee(name : String, id : Integer,
                                 hireDate : String, monthlySalary : Double)
        CALL Employee(name, id, hireDate)
        SET monthlySalary = monthlySalary

    METHODS
        calculateMonthlyPay() : Double
            RETURN monthlySalary
END CLASS


CLASS HourlyEmployee INHERITS Employee
    ATTRIBUTES
        hourlyRate : Double
        hoursWorked : Double

    CONSTRUCTOR HourlyEmployee(name : String, id : Integer,
                               hireDate : String,
                               hourlyRate : Double,
                               hoursWorked : Double)
        CALL Employee(name, id, hireDate)
        SET hourlyRate = hourlyRate
        SET hoursWorked = hoursWorked

    METHODS
        calculateMonthlyPay() : Double
            RETURN hourlyRate * hoursWorked
END CLASS


CLASS CommissionEmployee INHERITS Employee
    ATTRIBUTES
        basePay : Double
        sales : Double
        commissionRate : Double

    CONSTRUCTOR CommissionEmployee(name : String, id : Integer,
                                   hireDate : String,
                                   basePay : Double,
                                   sales : Double,
                                   commissionRate : Double)
        CALL Employee(name, id, hireDate)
        SET basePay = basePay
        SET sales = sales
        SET commissionRate = commissionRate

    METHODS
        calculateMonthlyPay() : Double
            RETURN basePay + (sales * commissionRate)
END CLASS


CLASS PayrollSystem
    ATTRIBUTES
        employees : List<Employee>

    CONSTRUCTOR PayrollSystem()
        CREATE empty employees list

    METHODS
        addEmployee(employee : Employee)
            ADD employee TO employees

        displayPayroll()
            FOR EACH employee IN employees
                PRINT employee.name
                PRINT employee.calculateMonthlyPay()
            END FOR
END CLASS


START PROGRAM

DECLARE payroll : PayrollSystem
SET payroll = NEW PayrollSystem()

DECLARE emp1 : SalariedEmployee
SET emp1 = NEW SalariedEmployee("Ahmed", 101, "2023-01-10", 5000)

DECLARE emp2 : HourlyEmployee
SET emp2 = NEW HourlyEmployee("Sara", 102, "2024-03-15", 20, 160)

DECLARE emp3 : CommissionEmployee
SET emp3 = NEW CommissionEmployee("Ali", 103, "2022-08-20", 3000, 10000, 0.10)

CALL payroll.addEmployee(emp1)
CALL payroll.addEmployee(emp2)
CALL payroll.addEmployee(emp3)

CALL payroll.displayPayroll()

END PROGRAM





Task 295: Zoo Animal Enclosures


CLASS Animal
    ATTRIBUTES
        name : String
        age : Integer
        dietType : String

    CONSTRUCTOR Animal(name : String, age : Integer, dietType : String)
        SET this.name = name
        SET this.age = age
        SET this.dietType = dietType

    METHODS
        makeSound()
            PRINT "Animal makes a sound"

        move()
            PRINT "Animal moves"
END CLASS


CLASS Bird INHERITS Animal
    ATTRIBUTES
        canFly : Boolean
        laysEggs : Boolean

    CONSTRUCTOR Bird(name : String, age : Integer,dietType : String,canFly : Boolean,laysEggs : Boolean)

        CALL SUPER(name, age, dietType)
        SET this.canFly = canFly
        SET this.laysEggs = laysEggs

    OVERRIDE METHOD makeSound()
        PRINT "Bird chirps"

    OVERRIDE METHOD move()
        PRINT "Bird flies"
END CLASS


CLASS Fish INHERITS Animal
    ATTRIBUTES
        swims : Boolean
        hasGills : Boolean

    CONSTRUCTOR Fish(name : String, age : Integer,dietType : String,swims : Boolean,hasGills : Boolean)
        CALL SUPER(name, age, dietType)
        SET this.swims = swims
        SET this.hasGills = hasGills

    OVERRIDE METHOD makeSound()
        PRINT "Fish makes bubbling sound"

    OVERRIDE METHOD move()
        PRINT "Fish swims"
END CLASS


CLASS Mammal INHERITS Animal
    ATTRIBUTES
        hasFur : Boolean
        givesLiveBirth : Boolean

    CONSTRUCTOR Mammal(name : String, age : Integer,dietType : String, hasFur : Boolean, givesLiveBirth : Boolean)
        CALL SUPER(name, age, dietType)
        SET this.hasFur = hasFur
        SET this.givesLiveBirth = givesLiveBirth

    OVERRIDE METHOD makeSound()
        PRINT "Mammal makes sound"

    OVERRIDE METHOD move()
        PRINT "Mammal walks"
END CLASS


CLASS Enclosure
    ATTRIBUTES
        temperatureSetting : Double
        animals : List(Animal)

    CONSTRUCTOR Enclosure(temperatureSetting : Double)
        SET this.temperatureSetting = temperatureSetting
        CREATE empty animals list

    METHODS
        addAnimal(animal : Animal)
            ADD animal TO animals

        displayAnimals()
            repate animal IN animals
                PRINT animal.name
                CALL animal.makeSound()
                CALL animal.move()
            END FOR
END CLASS


START PROGRAM

DECLARE enclosure1 : Enclosure
SET enclosure1 = NEW Enclosure(25.5)

DECLARE bird1 : Bird
SET bird1 = NEW Bird("Parrot", 2, "Herbivore", TRUE, TRUE)

DECLARE fish1 : Fish
SET fish1 = NEW Fish("Goldfish", 1, "Omnivore", TRUE, TRUE)

DECLARE mammal1 : Mammal
SET mammal1 = NEW Mammal("Lion", 5, "Carnivore", TRUE, TRUE)


CALL enclosure1.addAnimal(bird1)
CALL enclosure1.addAnimal(fish1)
CALL enclosure1.addAnimal(mammal1)
CALL enclosure1.displayAnimals()


END PROGRAM
